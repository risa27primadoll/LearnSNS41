$start = ($page -1)*5;


//最初の5件を取得
SELECT * FROM `feeds` ORDER BY `created` DESC LIMIT 0,5

//LIMIT句の使い方

//次の5件を取得
SELECT * FROM `feeds` ORDER BY `created` DESC LIMIT ５,5
